Функция рткПроверитьРасхожденияПередПроведением() 
	
	// ++   
	
	Результат = Новый Структура;
	Результат.Вставить("ЕстьРасхождения", Ложь);
	Результат.Вставить("Расхождения", Новый Массив);
	Результат.Вставить("ОбщееРасхождение", 0);
	
	Если Объект.ЗаказНаПеремещение = Неопределено Тогда
		Возврат Результат;
	КонецЕсли;
	
	МассивЗаказов = рткМассивЗаказов();
	
	ДанныеЗаказа = ПолучитьДанныеТЧЗаказаНаПеремещение(МассивЗаказов);
	
	//ДанныеПеремещения = ПолучитьДанныеТЧПеремещенияТоваров(ДокументПеремещения);
	ДанныеПеремещения = Объект.Товары.Выгрузить();
	
	// Создаем словари для сравнения
	СловарьЗаказа = СоздатьСловарьПоНоменклатуре(ДанныеЗаказа);
	СловарьПеремещения = СоздатьСловарьПоНоменклатуреПеремещ(ДанныеПеремещения);
	
	Для Каждого КлючЗначение Из СловарьЗаказа Цикл
		Ключ = КлючЗначение.Ключ;
		ДанныеЗаказаТЧ = КлючЗначение.Значение;
		
		ДанныеПеремещенияТЧ = СловарьПеремещения.Получить(Ключ);
		
		Если ДанныеПеремещенияТЧ = Неопределено Тогда
			ДобавитьРасхождение(Ключ, Результат, ДанныеЗаказаТЧ, Неопределено);
		Иначе
			Если ДанныеЗаказаТЧ.Количество <> ДанныеПеремещенияТЧ.Количество Тогда
				ДобавитьРасхождение(Ключ, Результат, ДанныеЗаказаТЧ, ДанныеПеремещенияТЧ);
			КонецЕсли;
			СловарьПеремещения.Удалить(Ключ);
		КонецЕсли;
	КонецЦикла;

	Для Каждого КлючЗначение Из СловарьПеремещения Цикл
		ДанныеПеремещенияТЧ = КлючЗначение.Значение;
		ДобавитьРасхождение(КлючЗначение.Ключ,Результат, Неопределено, ДанныеПеремещенияТЧ);
	КонецЦикла;
	
	// Отбор сверх заказа
	СтруктураОтбораСтроки = Новый Структура;
	СтруктураОтбораСтроки.Вставить("КодСтроки", 0);
	СверхЗаказа = Объект.Товары.НайтиСтроки(СтруктураОтбораСтроки); 
	
	Если СверхЗаказа.Количество() Тогда
		// Создаем словарь уже существующих расхождений по GUID
		СуществующиеРасхождения = Новый Соответствие;
		Для Каждого СущРасх Из Результат.Расхождения Цикл
			Если СущРасх.Свойство("УникальныйИдентификатор") Тогда
				КлючРасхождения = СущРасх.УникальныйИдентификатор;
				СуществующиеРасхождения.Вставить(КлючРасхождения, Истина);
			КонецЕсли;
		КонецЦикла;
		
		Для Каждого СтрокаСверх Из СверхЗаказа Цикл
			Если НЕ СтрокаСверх.ЗаказНаПеремещение = Документы.ЗаказНаПеремещение.ПустаяСсылка() Тогда
				ГУИДНоменклатурыСверх = Строка(СтрокаСверх.Номенклатура.УникальныйИдентификатор());
				ГУИДХарактеристикиСверх = "БезХарактеристики";
				Если ЗначениеЗаполнено(СтрокаСверх.Характеристика) Тогда
					ГУИДХарактеристикиСверх = Строка(СтрокаСверх.Характеристика.УникальныйИдентификатор());
				КонецЕсли;
				
				КлючПроверки = ГУИДНоменклатурыСверх + "_" + ГУИДХарактеристикиСверх + "_" + СтрокаСверх.ЗаказНаПеремещение.Номер;
				
				// Добавляем расхождение только если его еще нет  
				НашлиСверхРасхождения = СуществующиеРасхождения.Получить(КлючПроверки);
				
				Если НашлиСверхРасхождения = Неопределено Тогда
					ДобавитьРасхождениеСверхПеремещения(СтрокаСверх, Результат);
					СуществующиеРасхождения.Вставить(КлючПроверки, Истина);
				КонецЕсли;
			Иначе 
				Продолжить;
			КонецЕсли;
		КонецЦикла;
	КонецЕсли;
		
	Результат.ЕстьРасхождения = Результат.Расхождения.Количество() > 0;
	
	Возврат Результат;   
	
	// --

Процедура ДобавитьРасхождение(Ключ, Результат, ДанныеЗаказа, ДанныеПеремещения)
	
	// ++  
	
	Расхождение = Новый Структура;
	
	Если ДанныеЗаказа = Неопределено Тогда
		// Лишняя позиция в перемещении
		Расхождение.Вставить("Тип", "ЛишняяПозиция");
		Расхождение.Вставить("Артикул", ДанныеПеремещения.Артикул);
		Расхождение.Вставить("Наименование", ДанныеПеремещения.Наименование);
		Расхождение.Вставить("Характеристика", ?(ДанныеПеремещения.Характеристика = Неопределено, 
											   "Без характеристики", 
											   ДанныеПеремещения.Характеристика.Наименование));
		Расхождение.Вставить("КоличествоЗаказано", 0);
		Расхождение.Вставить("КоличествоПеремещается", ДанныеПеремещения.Количество);
		Расхождение.Вставить("Разница", -ДанныеПеремещения.Количество);
		Расхождение.Вставить("УникальныйИдентификатор", Ключ);   
		
	ИначеЕсли ДанныеПеремещения = Неопределено Тогда
		// Отсутствующая позиция в перемещении
		Расхождение.Вставить("Тип", "ОтсутствующаяПозиция");
		Расхождение.Вставить("Артикул", ДанныеЗаказа.Артикул);
		Расхождение.Вставить("Наименование", ДанныеЗаказа.Наименование);
		Расхождение.Вставить("Характеристика", ?(ДанныеЗаказа.Характеристика = Неопределено, 
											   "Без характеристики", 
											   ДанныеЗаказа.Характеристика.Наименование));
		Расхождение.Вставить("КоличествоЗаказано", ДанныеЗаказа.Количество);
		Расхождение.Вставить("КоличествоПеремещается", 0);
		Расхождение.Вставить("Разница", ДанныеЗаказа.Количество);
		Расхождение.Вставить("УникальныйИдентификатор", Ключ);
		Расхождение.Вставить("Номер", ДанныеЗаказа.Номер);

	Иначе
		// Разное количество
		Расхождение.Вставить("Тип", "РазноеКоличество");
		Расхождение.Вставить("Артикул", ДанныеЗаказа.Артикул);
		Расхождение.Вставить("Наименование", ДанныеЗаказа.Наименование);
		Расхождение.Вставить("Характеристика", ?(ДанныеЗаказа.Характеристика = Неопределено, 
											   "Без характеристики", 
											   ДанныеЗаказа.Характеристика.Наименование));
		Расхождение.Вставить("КоличествоЗаказано", ДанныеЗаказа.Количество);
		Расхождение.Вставить("КоличествоПеремещается", ДанныеПеремещения.Количество);
		Расхождение.Вставить("Разница", ДанныеЗаказа.Количество - ДанныеПеремещения.Количество);
		Расхождение.Вставить("УникальныйИдентификатор", Ключ); 
		Расхождение.Вставить("Номер", ДанныеЗаказа.Номер);

	КонецЕсли;
	
	Результат.Расхождения.Добавить(Расхождение);
	Результат.ОбщееРасхождение = Результат.ОбщееРасхождение + ?(Расхождение.Разница > 0, Расхождение.Разница, -Расхождение.Разница);  
	
	// --

Функция ОбъектМодифицирован(ПроверятьТабличныеЧасти = Истина) Экспорт
	
	// ++
	
	Ссылка = Объект.Ссылка;
	МетаданныеОбъекта = Ссылка.Метаданные();
	
	Если ПроверятьТабличныеЧасти Тогда
		Для каждого ТЧ Из МетаданныеОбъекта.ТабличныеЧасти Цикл
			
			Если Объект[ТЧ.Имя].Количество() <> Ссылка[ТЧ.Имя].Количество() Тогда
				Возврат Истина;
			КонецЕсли;
			
			Для Индекс = 0 По Объект[ТЧ.Имя].Количество()-1 Цикл
				ОбъектСтрокаТЧ = Объект[ТЧ.Имя][Индекс];
				СсылкаСтрокаТЧ = Ссылка[ТЧ.Имя][Индекс];
				Для каждого РеквизитТЧ Из ТЧ.Реквизиты Цикл
					Если ОбъектСтрокаТЧ[РеквизитТЧ.Имя] <> СсылкаСтрокаТЧ[РеквизитТЧ.Имя] Тогда
						Возврат Истина;
					КонецЕсли;
				КонецЦикла;
			КонецЦикла;
			
		КонецЦикла;        
	КонецЕсли;
	
    Возврат Ложь;   
	
	// --
